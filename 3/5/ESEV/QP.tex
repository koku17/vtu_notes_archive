% macro
\def \qa#1#2#3#4#5#6{
	\item #1
	\begin{enumerate}[label=\alph*)]
		\begin{minipage}{.5\textwidth}
			\item #2
			\item #4
		\end{minipage}
		\hfill
		\begin{minipage}{.5\textwidth}
			\item #3
			\item #5
		\end{minipage}
	\end{enumerate}
	\item[$\to$]
		\if#6a a) #2 \fi
		\if#6b b) #3 \fi
		\if#6c c) #4 \fi
		\if#6d d) #5 \fi
}

\def \lqa#1#2#3#4#5#6{
	\item #1
		\begin{enumerate}[label=\alph*)]
			\item #2
			\item #3
			\item #4
			\item #5
		\end{enumerate}
	\item[$\to$]
		\if#6a a) #2 \fi
		\if#6b b) #2 \fi
		\if#6c c) #2 \fi
		\if#6d d) #2 \fi
}
